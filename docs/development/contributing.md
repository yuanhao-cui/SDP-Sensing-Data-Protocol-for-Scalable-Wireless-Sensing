```bash
git clone https://github.com/yuanhao-cui/SDP-Sensing-Data-Protocol-for-Scalable-Wireless-Sensing.git
cd SDP-Sensing-Data-Protocol-for-Scalable-Wireless-Sensing
pip install -e ".[dev]"
pytest tests/ -v
```
